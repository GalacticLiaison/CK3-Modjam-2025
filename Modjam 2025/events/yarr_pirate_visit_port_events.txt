			}
			triggered_desc = {
				trigger = {
					scope:activity = {

							category = special_type
							option = pirate_visit_port_type_gambling
						}
					}
				}

			}
		}
	}

	left_portrait = {
		character = root
		animation = personality_greedy
	}

	theme = travel
	override_background = { reference = docks }
